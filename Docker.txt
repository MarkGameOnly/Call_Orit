# Используем официальный образ Python
FROM python:3.9-slim-buster

# Устанавливаем рабочую директорию в контейнере
WORKDIR /app

# Копируем файл requirements.txt и устанавливаем зависимости
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Копируем все остальные файлы приложения
COPY . .

# Создаем директорию для изображений, если она не существует
RUN mkdir -p img

# Копируем изображения в контейнер (убедитесь, что папка img существует локально)
# Если у вас нет изображений, можете закомментировать эту строку
COPY img/ img/

# Открываем порт, на котором будет работать FastAPI
EXPOSE 8000

# Команда для запуска приложения с Uvicorn
CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]
